\footnotesize
\noindent\textbf{Output format.}
\code{\{"fit": "Yes"|"No", "motivation": "1--2 sentence reason"\}}.

\smallskip\noindent\textbf{Profiles.}
$24$ sets (income $\times$ employment $\times$ marital $\times$ children)
$\times$ $20$ applicants ($2$ genders $\times$ $5$ backgrounds $\times$ $2$ reps
$=480$). Repository:
\url{https://github.com/Sam9875/Tenant-bias-LLM}.

\smallskip\noindent\textbf{Baseline prompt.}
Landlord-assistant role; rental announcement (title, description, rent, size,
bedrooms, neighbourhood, furnished); houseseeker profile (gender, background,
income, employment, marital status, children); JSON-only reply.

\smallskip\noindent\textbf{RQ4 variants.}
\emph{Fairness:} same template plus a rule to ignore gender, ethnicity,
immigration, and refugee status. \emph{CoT:} same template plus three-step
internal reasoning before the JSON answer.

\smallskip\noindent\textbf{Example profile (S14\_musy\_refugee\_male\_1).}
Medium income (\eur28{,}000), unemployed, single parent with one child, male,
refugee/asylum-seeker.
